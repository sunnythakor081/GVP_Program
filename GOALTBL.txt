    @Override
    public ApiResponseModel<?> getTeamMemberList(
            CirightOKRTeamMemberModel model) throws CirightException {

        if (model.getSubscriptionId() == null
                || model.getSubscriptionId().longValue() == -1) {
            throw new InvalidParameterException(
                    "Subscription Id is not available", "subscriptionId");
        }

        if (model.getVerticalId() == null
                || model.getVerticalId().longValue() == -1) {
            throw new InvalidParameterException("Vertical Id is not available",
                    "verticalId");
        }

        if (model.getAppId() == null || model.getAppId().longValue() == -1) {
            throw new InvalidParameterException("App Id is not available",
                    "appId");
        }

        if (model.getLoginEmployeeId() == null
                || model.getLoginEmployeeId().longValue() == -1) {
            throw new InvalidParameterException("Employee Id is not available",
                    "loginEmployeeId");
        }

        final ApiResponseModel<List<CirightOKRTeamMemberModel>> apiResponse = new ApiResponseModel<>();

        final Map<String, Object> options = new HashMap<>();

        options.put(UtilsQueryParameterConstants.SUBSCRIPTION_ID,
                model.getSubscriptionId());
        options.put(UtilsQueryParameterConstants.VERTICAL_ID,
                model.getVerticalId());

        if (model.getIsLead() != null) {
            options.put("isLead", model.getIsLead());
        }

        String sqlQuery = "SELECT teat.active_team_id AS col_0_0, GROUP_CONCAT(teat.team_id SEPARATOR ', ') AS col_0_1,"
                + " GROUP_CONCAT(tet.Team_Description SEPARATOR ', ') AS col_0_2, teat.employee_id AS col_0_3,"
                + " CONCAT(te_emp.FirstName, ' ', te_emp.LastName) AS col_0_4, teat.`role` AS col_0_5,"
                + " IF(teat.status = 1, 'Active', 'Inactive') AS col_0_6,"
                + " MAX(CASE WHEN tet.team_lead = teat.employee_id THEN 1 ELSE 0 END) AS is_team_lead,"
                + " COUNT(DISTINCT CASE WHEN tg.Manager = teat.employee_id THEN tg.GoalID END) AS total_goals_owned"
                + " FROM tbl_employee_active_team teat"
                + " INNER JOIN tbl_employee te ON te.EmployeeID = teat.created_by"
                + " INNER JOIN tbl_employee_team tet ON tet.TeamId = teat.team_id AND tet.Team_Active = 1"
                + " INNER JOIN tbl_employee te_emp ON te_emp.EmployeeID = teat.employee_id AND te_emp.Active_Employee = 1"
                + " LEFT JOIN tbl_goals tg ON tet.TeamId = tg.TeamId AND tg.IsActive = 1 AND tg.Manager = teat.employee_id"
                + " WHERE te.CompanyID = :subscriptionId AND te.Vertical_Id = :verticalId"
                + " GROUP BY teat.employee_id, teat.`role`, CONCAT(te_emp.FirstName, ' ', te_emp.LastName), teat.status"
                + (model.getIsLead() != null && Short.valueOf(
                        model.getIsLead()) == UtilsConfigConstants.ACTIVE_STATUS
                                ? " HAVING is_team_lead = 1 "
                                : "")
                + (model.getIsLead() != null && Short.valueOf(model
                        .getIsLead()) == UtilsConfigConstants.INACTIVE_STATUS
                                ? " HAVING is_team_lead = 0 "
                                : "")
                + " ORDER BY MAX(teat.created_date) DESC";

        final List<Object[]> resultList = getFacadeLookup()
                .getPersistenceFacade().findByNativeQueryC(sqlQuery,
                        new QueryOptions(options), Object[].class);

        List<CirightOKRTeamMemberModel> responseModel = new ArrayList<>();

        if (resultList != null && !resultList.isEmpty()) {

            for (final Object[] obj : resultList) {

                CirightOKRTeamMemberModel teamMemberModel = new CirightOKRTeamMemberModel();

                teamMemberModel.setActiveTeamMemberId(
                        obj[0] != null ? Long.valueOf(obj[0].toString()) : -1L);
                teamMemberModel.setTeamIds(
                        obj[1] != null && !obj[1].toString().isEmpty()
                                ? Arrays.stream(obj[1].toString().split(","))
                                        .map(String::trim).map(Long::valueOf)
                                        .toArray(Long[]::new)
                                : new Long[0]);

                teamMemberModel.setTeams(
                        obj[2] != null && !obj[2].toString().isEmpty() ? Arrays
                                .stream(obj[2].toString().split(","))
                                .map(String::trim).toArray(String[]::new)
                                : new String[0]);
                teamMemberModel.setEmployeeId(
                        obj[3] != null ? Long.valueOf(obj[3].toString()) : -1L);
                teamMemberModel.setEmployeeName(
                        obj[4] != null ? obj[4].toString() : "");
                teamMemberModel
                        .setRole(obj[5] != null ? obj[5].toString() : "");
                teamMemberModel
                        .setStatus(obj[6] != null ? obj[6].toString() : "");
                teamMemberModel.setIsLead(
                        obj[7] != null ? Short.valueOf(obj[7].toString())
                                : UtilsConfigConstants.INACTIVE_STATUS);
                teamMemberModel.setObjectivesAssignedCount(
                        obj[8] != null ? Integer.valueOf(obj[8].toString())
                                : 0);

                responseModel.add(teamMemberModel);

            }
        }
        apiResponse.setStatus(true);
        apiResponse.setMessage("success");
        apiResponse.setData(responseModel);
        return apiResponse;
    }

    public ApiResponseModel<?> updateMembersTeam(
            CirightOKRTeamMemberModel model) throws InvalidParameterException,
            CirightException, FieldAlreadyExistException {

        if (model.getEmployeeId() == null
                || model.getEmployeeId().longValue() == -1) {
            throw new InvalidParameterException("Employee Id is not available",
                    "employeeId");
        }

        if (model.getTeamIds() == null) {
            throw new InvalidParameterException("Team Ids is not available",
                    "teamIds");
        }

        if (model.getLoginEmployeeId() == null
                || model.getLoginEmployeeId().longValue() == -1) {
            throw new InvalidParameterException("Login Employee Id is not available",
                    "loginEmployeeId");
        }
        
        final ApiResponseModel<?> apiResponse = new ApiResponseModel<>();
        final Map<String, Object> options = new HashMap<>();

        List<Long> resultList = null;
        Set<Long> currentIds = null;

        if (model.getRole() != null && !model.getRole().equals("")) {

            options.put(UtilsQueryParameterConstants.EMPLOYEE_ID,
                    model.getEmployeeId());
            options.put("role", model.getRole());

            resultList = getFacadeLookup().getPersistenceFacade()
                    .findByNamedQueryC(
                            "get.active.team.id.by.employee.id.and.role",
                            new QueryOptions(options));

            currentIds = new HashSet<>(resultList);

        } else {

            options.clear();
            options.put(UtilsQueryParameterConstants.EMPLOYEE_ID,
                    model.getEmployeeId());

            resultList = getFacadeLookup().getPersistenceFacade()
                    .findByNamedQueryC("get.active.team.id.by.employee.id",
                            new QueryOptions(options));

            currentIds = new HashSet<>(resultList);

        }

        Set<Long> incomingIds = new HashSet<>(
                Arrays.asList(model.getTeamIds()));

        /* ------- To Add Entry */

        Set<Long> toAdd = new HashSet<>(incomingIds);
        toAdd.removeAll(currentIds);

        if (!toAdd.isEmpty()) {

            final TblEmployeeActiveTeam tblEmployeeActiveTeam = new TblEmployeeActiveTeam();

            tblEmployeeActiveTeam.setCreatedDate(new Date());

            final TblEmployee tblEmployeeCreatedBy = new TblEmployee();
            tblEmployeeCreatedBy.setId(model.getLoginEmployeeId());
            tblEmployeeActiveTeam
                    .setTblEmployeeByCreatedBy(tblEmployeeCreatedBy);

            tblEmployeeActiveTeam.setStatus(UtilsConfigConstants.ACTIVE_STATUS);

            final TblEmployee tblEmployee = new TblEmployee();
            tblEmployee.setId(model.getEmployeeId());
            tblEmployeeActiveTeam.setTblEmployee(tblEmployee);

            if (model.getNewRole() != null && !model.getNewRole().equals("")) {
                tblEmployeeActiveTeam.setRole(model.getNewRole());
            }

            for (Long teamId : toAdd) {

                final TblEmployeeTeam tblEmployeeTeam = new TblEmployeeTeam();
                tblEmployeeTeam.setId(teamId);
                tblEmployeeActiveTeam.setTblEmployeeTeam(tblEmployeeTeam);

                getFacadeLookup().getPersistenceFacade()
                        .update(tblEmployeeActiveTeam);

            }
        }

        /* ------- To Update Entries */

        Set<Long> toRemove = new HashSet<>(currentIds);
        toRemove.removeAll(incomingIds);

        if (!toRemove.isEmpty()) {

            options.clear();
            options.put("activeTeamMemberIds", toRemove);
            options.put(UtilsQueryParameterConstants.NEW_STATUS,
                    UtilsConfigConstants.INACTIVE_STATUS);

            getFacadeLookup().getPersistenceFacade().findByNamedExecuteQueryC(
                    "update.team.member.status.by.id",
                    new QueryOptions(options));
        }

        apiResponse.setStatus(true);
        apiResponse.setMessage("success");
        apiResponse.setData(null);
        return apiResponse;

    }
